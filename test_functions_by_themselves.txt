def test_can_access_google_page_accessible(
    monkeypatch: pytest.MonkeyPatch,
) -> None:
    monkeypatch.setattr(
        main,
        "has_internet_connection",
        lambda: True,
    )
    monkeypatch.setattr(main, "valid_google_url", lambda url: True)
    result = main.can_access_google_page("https://www.google.com")
    assert result == "Accessible"


def test_can_access_google_page_no_internet(
    monkeypatch: pytest.MonkeyPatch,
) -> None:
    # Mock: sin conexión a internet
    monkeypatch.setattr(main, "has_internet_connection", lambda: False)

    # Mock: url válida (no debería influir)
    monkeypatch.setattr(main, "valid_google_url", lambda url: True)
    result = main.can_access_google_page("https://www.google.com")
    assert result == "Not accessible"


def test_can_access_google_page_no_internet_does_not_call_valid_url(
    monkeypatch: pytest.MonkeyPatch,
) -> None:
    monkeypatch.setattr(main, "has_internet_connection", lambda: False)

    valid_mock = Mock(return_value=True)
    monkeypatch.setattr(main, "valid_google_url", valid_mock)
    result = main.can_access_google_page("https://www.google.com")
    assert result == "Not accessible"
    valid_mock.assert_not_called()
